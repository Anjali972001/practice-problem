public class Solution {
	
	public static long maxSubarraySum(int[] arr, int n) {
		// write your code here
        long maxSum=0;
        long curSum=0;
        for(int i=0;i<n;i++)
        {
            curSum=curSum+arr[i];
                if(maxSum<curSum)
                {
                    maxSum=curSum;
                }
            if(curSum<0)
            {
                curSum=0;
            }
            
        }
        return maxSum;
	}

}
